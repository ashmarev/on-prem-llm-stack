## vLLM
```
services:
  vllm:
    image: vllm/vllm-openai:latest
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: "all"
              capabilities: [gpu]
    container_name: vllm
    ports:
      - "8000:8000"
    volumes:
      - ~/.cache/huggingface:/root/.cache/huggingface
    entrypoint: ["python3", "-m", "vllm.entrypoints.openai.api_server"]
    command:
      - --model 
      - Qwen/Qwen3-4B-Instruct-2507-FP8
      - --host
      - 0.0.0.0
      - --port
      - "8000"
      - --max-model-len
      - "16384"
      - --gpu-memory-utilization
      - "0.95"
```
* Запускал не через докер, обрати внимание, что max-model-len и max-num-seqs пришлость поджать
```
vllm serve Qwen/Qwen3-4B-Instruct-2507-FP8 --gpu-memory-utilization 0.95 --max-model-len 32768 --max-num-seqs 64
```
## vLLM bench
```
for conc in 1 2 4 8 16 32; do
  echo "=== Concurrency $conc ==="
  vllm bench serve \
    --backend openai \
    --model Qwen/Qwen3-4B-Instruct-2507-FP8 \
    --dataset-name random \
    --num-prompts 64 \
    --random-input-len 400 \
    --random-output-len 150 \
    --request-rate inf \
    --max-concurrency $conc \
    --ignore-eos \
    --percentile-metrics ttft,itl,tpot,e2el \
    --metric-percentiles 50,90,95,99 \
    --save-result \
    --save-detailed \
    --result-filename result-conc-$conc.json
  sleep 10
done
```
